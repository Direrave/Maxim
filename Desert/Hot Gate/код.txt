// Используй голосовые команды для управления артиллерией.
while(true) {
	var enemy = pet.findNearestEnemy();
	if(!enemy) {
		continue;
	}
//Разведчики быстры. Нужно их остановить.
	if(enemy.type == "scout") {
		var distance = pet.distanceTo(enemy);
		if(pet.isReady("coldBlast") && distance < 5) {
			pet.coldBlast();
		}
	}
	else {
//Если враг слева от питомца:
		if(enemy.pos.x < pet.pos.x) {
// Скажи "left"
			pet.say("left");
		}
//Если справа от питомца:
		if(enemy.pos.x > pet.pos.x) {
			pet.say("right");
		}
	}
}
