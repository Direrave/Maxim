// Не позволь ограм вступить в рощу.
function onSpawn() {
	while (true) {
		var scout = pet.findNearestByType("scout");
		if (scout && pet.isReady("charm")) {
			pet.charm(scout);
		}
	}
}

//Назначь обработчик событий на событие питомца "spawn".
pet.on("spawn", onSpawn);
