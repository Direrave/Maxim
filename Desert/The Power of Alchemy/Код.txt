// Жди команды алхимика, чтобы собрать зелья.
// Обработчик для события питомца "hear",
function onHear(event) {
//Найди ближайшее зелье.
	var potion = pet.findNearestByType("potion");
	var message = event.message;
//Если сообщение события - "Fetch!"
	if (message == "Fetch") {
//Питомец должен взять зелье
		pet.fetch(potion);
//Иначе (для любого другого сообщения):
	} else if(message != "Fetch") {
			pet.moveXY(54, 34);
	}
}

pet.on("hear", onHear);
