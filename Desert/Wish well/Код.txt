//Эта функция вычисляет сумарную стоимость монет.
function sumCoinValues(coins) {
	var coinIndex = 0;
	var totalValue = 0;
//Перебери все монеты.
	while(coinIndex < coins.length) {
		totalValue += coins[coinIndex].value;
		coinIndex ++;
	}
	return totalValue;
}
