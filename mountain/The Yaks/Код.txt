function removeByType(enemies, excludedType) {
	var tempArray = [];
	for(var i = 0; i < enemies.length; i++) {
		enemy = enemies[i];
		if(enemy.type != excludedType) {
			tempArray.push(enemy);
		}
	}
	return tempArray;
}
